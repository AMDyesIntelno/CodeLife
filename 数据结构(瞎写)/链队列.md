```c++
#include<iostream>
using namespace std;
typedef struct queue{
    int data;
    struct queue *next=NULL;
}*queueptr;
void queue_create(queueptr &p){
    p=new queue;
    p->data=0;
    return;
}
void queue_clear(queueptr base,queueptr &top){
    queueptr temp=base->next;
    while(temp){
        top=temp->next;
        delete temp;
        temp=top;
    }
    top=base;
    base->next=NULL;
    base->data=0;
    return;
}
void queue_destory(queueptr &base,queueptr &top){
    while(base){
        top=base->next;
        delete base;
        base=top;
    }
    base=NULL;
    top=NULL;
    return;
}
bool is_queue_empty(queueptr base,queueptr top){
    if(top==base){
        return 1;
    }
    else{
        return 0;
    }
}
int queue_size(queueptr base){
    return base->data;
}
void queue_push(queueptr base,queueptr &top,int item){
    queueptr temp=new queue;
    temp->data=item;
    top->next=temp;
    top=temp;
    base->data++;
    return; 
}
int queue_pop(queueptr base){
    if(base->data>0){
        queueptr temp=base->next->next;
        delete base->next;
        base->next=temp;
        return 1;
    }
    return 0;
}
int queue_top(queueptr base){
    return base->next->data;
}
void queue_show(queueptr base){
    queueptr temp=base->next;
    while(temp){
        cout<<temp->data<<' ';
        temp=temp->next;
    }
    cout<<endl;
    return;
}
int main(){
    queueptr base=NULL;
    queue_create(base);
    queueptr top=base;
    for(int i=1;i<=5;++i){
        queue_push(base,top,i*i);
    }
    queue_show(base);
    queue_pop(base);
    queue_pop(base);
    queue_pop(base);
    queue_show(base);
    queue_push(base,top,100);
    queue_push(base,top,200);
    queue_push(base,top,300);
    queue_show(base);
    cout<<"top:"<<queue_top(base)<<" size:"<<queue_size(base)<<endl;
    queue_clear(base,top);
    queue_show(base);
    for(int i=3;i<=6;++i){
        queue_push(base,top,i*i);
    }
    queue_show(base);
    queue_destory(base,top);
    return 0;
}
```
