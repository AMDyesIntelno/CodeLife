***Sample***

`46+5*(120-37)`

***Out***
```
46 5 120 37 - * + 
461
```


```c++
//中缀表达式转逆波兰式
#include<iostream>
#include<string>
#include<stack>
using namespace std;
int fir(char c){//比较加减和乘除优先级
    if(c=='+'||c=='-'){
        return 0;
    }
    else{
        return 1;
    }
}
void Transform(string &str,stack <string>&s2){//将中缀表达式转换成逆波兰式
    stack <char>symbol;//临时符号栈
    string temp;
    for(int i=0;i<str.size();++i){
        if(str[i]<='9'&&str[i]>='0'){//数字处理,将中缀表达式中的符号和数字分离
            temp+=str[i];
        }
        else if(str[i]=='('){//当遇到左括号时，将数字压入处理栈中，将左括号压入临时符号栈中
            symbol.push(str[i]);
            s2.push(temp);
            temp.erase();
        }
        else if(str[i]==')'){//当遇到右括号时，将数字压入处理栈中，此时，右括号不必压入临时符号栈
            s2.push(temp);
            temp.erase();
            while(symbol.top()!='('){//寻找离栈顶最近的一个左括号,即处理一对括号中的运算
                //此时，弹出临时符号栈中的符号并压入处理栈中
                temp+=symbol.top();
                s2.push(temp);
                symbol.pop();
                temp.erase();
            }
            symbol.pop();//将左括号丢弃,逆波兰式中不能出现括号
        }
        else{//当遇到运算符时，将数字压入处理栈中
            s2.push(temp);
            temp.erase();
            if(symbol.empty()){//当临时符号栈为空时，直接将当前运算符压入栈中
                symbol.push(str[i]);
            }
            else{//将当前运算符与临时符号栈的栈顶运算符进行比较
                if(fir(str[i])>fir(symbol.top())){//当前运算符的优先级高于(不包括等于)临时符号栈的栈顶运算符，继续压入临时符号栈中
                    //即当+-运算遇到*/运算时,继续压栈,暂不处理
                    symbol.push(str[i]);
                }
                else{
                    while(!symbol.empty()&&symbol.top()!='('&&fir(symbol.top())>=fir(str[i])){
                        /*弹出临时符号栈的栈顶元素，将其转换为string类型并压入处理栈中直到
                        1.临时符号栈为空
                        2.临时符号栈的栈顶元素为左括号
                        3.临时符号栈的栈顶元素的优先级低于(不包括等于)当前运算符的优先级*/
                        temp+=symbol.top();
                        symbol.pop();
                        s2.push(temp);
                        temp.erase();
                    }
                    symbol.push(str[i]);//将当前运算符压入临时符号栈
                }
            }
        }
    }
    s2.push(temp);//将最后一个数压入处理栈中,因为中缀表达式中的最后一个元素时数字,在for循环中无法直接处理
    temp.erase();//清空临时字符串
    while(!symbol.empty()){//将剩余的运算符压入处理栈中,在处理这一步时,显然不存在括号匹配和高优先级的运算符
        temp+=symbol.top();
        symbol.pop();
        s2.push(temp);
        temp.erase();
    }
    str.erase();//将中缀表达式清空,用于存放逆波兰式
    return;
}
int Calculate(string &str,stack <string>&s2){//计算逆波兰式
    stack <string>reverse;//逆向栈
    stack <int>num;//运算结果栈
    int t=0,num1=0,num2=0;
    while(!s2.empty()){//将处理栈逆序,若不进行逆向,将无法进行运算
        reverse.push(s2.top());
        s2.pop();
    }
    while(!reverse.empty()){//将reverse栈链接成逆波兰式，并利用reverse栈进行运算
        if(reverse.top()=="+"){
            num1=num.top();
            num.pop();
            num2=num.top();
            num.pop();
            num.push(num2+num1);
        }
        else if(reverse.top()=="-"){
            num1=num.top();
            num.pop();
            num2=num.top();
            num.pop();
            num.push(num2-num1);//注意是num2-num1
        }
        else if(reverse.top()=="*"){
            num1=num.top();
            num.pop();
            num2=num.top();
            num.pop();
            num.push(num2*num1);
        }
        else if(reverse.top()=="/"){
            num1=num.top();
            num.pop();
            num2=num.top();
            num.pop();
            num.push(num2/num1);//注意是num2/num1
        }
        else if(reverse.top().size()){//可能有空的栈单元
            for(int i=0;i<reverse.top().size();++i){//将栈中的数字字符串转换为数字,也可以用stoi
                t*=10;
                t+=reverse.top()[i]-'0';
            }
            num.push(t);
            t=0;
        }
        str+=reverse.top();//生成逆波兰式
        if(reverse.top().size()){
            str+=" ";
        }
        reverse.pop();
    }
    return num.top();
}
void Read(string &str){//读入中缀表达式
    cin>>str;
    return;
}
int main(){
    string str;//存放中缀表达式
    Read(str);
    stack <string>s2;//处理栈
    Transform(str,s2);
    int num=Calculate(str,s2);
    cout<<str<<endl<<num;
    return 0;
}
```
